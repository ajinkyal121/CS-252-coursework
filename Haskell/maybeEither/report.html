getMax :: [Int] -> Maybe Int
getMax [] = Nothing
getMax (x:xs) 
   | Just x > (getMax xs)     = Just x
   | otherwise                  = getMax xs
   



reciprocal :: (Eq a, Fractional a) => a -> Maybe a
reciprocal 0 = Nothing
reciprocal x = Just (1 / x)
